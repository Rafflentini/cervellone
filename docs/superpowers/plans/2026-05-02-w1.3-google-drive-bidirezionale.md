```ts
export async function searchFilesFullText(query: string, folderId?: string): Promise<string> {
  try {
    const drive = getDrive()
    let q = `fullText contains '${query.replace(/'/g, "\\'")}' and trashed = false`
    if (folderId) q += ` and '${folderId}' in parents`

    const res = await drive.files.list({
      q,
      fields: 'files(id, name, mimeType, parents, modifiedTime)',
      orderBy: 'modifiedTime desc',
      pageSize: 20,
    })

    const files = res.data.files || []
    if (files.length === 0) return `Nessun file con contenuto "${query}".`

    const lines = files.map(f => {
      const isFolder = f.mimeType === 'application/vnd.google-apps.folder'
      return `${isFolder ? '📁' : '📄'} ${f.name} [ID: ${f.id}]`
    })

    return `${files.length} file con contenuto "${query}":\n${lines.join('\n')}`
  } catch (err) {
    return `Errore nella ricerca full-text: ${err}`
  }
}
```

```ts
async function downloadFile(fileId: string): Promise<{ buffer: Buffer; mimeType: string; name: string }> {
  const drive = getDrive()
  const meta = await drive.files.get({ fileId, fields: 'name, mimeType, size' })
  const sizeBytes = Number(meta.data.size || 0)
  if (sizeBytes > 20 * 1024 * 1024) {
    throw new Error(`File troppo grande (${(sizeBytes / 1024 / 1024).toFixed(1)} MB > 20 MB max)`)
  }
  const res = await drive.files.get({ fileId, alt: 'media' }, { responseType: 'arraybuffer' })
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const buffer = Buffer.from(res.data as any)
  return {
    buffer,
    mimeType: meta.data.mimeType || 'application/octet-stream',
    name: meta.data.name || 'file',
  }
}
```

```ts
export async function readPdfFromDrive(fileId: string): Promise<string> {
  try {
    const file = await downloadFile(fileId)
    if (!file.mimeType.includes('pdf')) {
      return `File "${file.name}" non è un PDF (mime: ${file.mimeType}). Usa drive_read_office o drive_read_document.`
    }
    // eslint-disable-next-line @typescript-eslint/no-require-imports
    const pdfParse = (await import('pdf-parse')).default
    const data = await pdfParse(file.buffer)
    const text = (data.text || '').slice(0, 50000)
    return `📄 PDF: ${file.name} (${data.numpages} pagine)\n\n${text || '(testo vuoto o solo immagini — considera Vision OCR)'}`
  } catch (err) {
    return `Errore lettura PDF: ${err instanceof Error ? err.message : err}`
  }
}
```

```ts
export async function readDocxFromDrive(fileId: string): Promise<string> {
  try {
    const file = await downloadFile(fileId)
    const isDocx = file.mimeType.includes('wordprocessing') || file.name.toLowerCase().endsWith('.docx')
    if (!isDocx) return `File "${file.name}" non è un DOCX (mime: ${file.mimeType}).`
    const mammoth = await import('mammoth')
    const result = await mammoth.extractRawText({ buffer: file.buffer })
    return `📄 DOCX: ${file.name}\n\n${(result.value || '').slice(0, 50000)}`
  } catch (err) {
    return `Errore lettura DOCX: ${err instanceof Error ? err.message : err}`
  }
}
```

- [ ] **Step 2: readXlsxFromDrive**

```ts
export async function readXlsxFromDrive(fileId: string): Promise<string> {
  try {
    const file = await downloadFile(fileId)
    const isXlsx = file.mimeType.includes('spreadsheet') || file.name.toLowerCase().endsWith('.xlsx')
    if (!isXlsx) return `File "${file.name}" non è un XLSX (mime: ${file.mimeType}).`
    const JSZip = (await import('jszip')).default
    const zip = await JSZip.loadAsync(file.buffer)
    const shared = await zip.file('xl/sharedStrings.xml')?.async('string')
    if (!shared) return `XLSX "${file.name}" senza sharedStrings — vuoto?`
    const texts: string[] = []
    const re = /<t[^>]*>([\s\S]*?)<\/t>/g
    let m
    while ((m = re.exec(shared)) !== null) texts.push(m[1].trim())
    return `📊 XLSX: ${file.name}\n\n${texts.slice(0, 5000).join(' | ').slice(0, 50000)}`
  } catch (err) {
    return `Errore lettura XLSX: ${err instanceof Error ? err.message : err}`
  }
}
```

```ts
/**
 * Cerca cantiere in REGISTRO_CANTIERI Sheet (header su 3 righe, dati da riga 4).
 * Match fuzzy su colonna "nome cliente" o "denominazione cantiere".
 * Ritorna folder ID se trovato, null altrimenti.
 */
export async function findCantiereFolder(query: string): Promise<{ folderId: string; folderName: string } | null> {
  try {
    const sheets = getSheets()
    const res = await sheets.spreadsheets.values.get({
      spreadsheetId: SHEETS.REGISTRO_CANTIERI,
      range: 'A4:M500', // adattare al range reale del registro
    })
    const rows = res.data.values || []
    if (!rows.length) return null

    const queryLower = query.toLowerCase()
    // Assume colonne: A=cliente, B=denominazione, ..., L=folder_id (adattare in base al sheet reale)
    for (const row of rows) {
      const cliente = String(row[0] || '').toLowerCase()
      const denom = String(row[1] || '').toLowerCase()
      const folderId = String(row[11] || '') // colonna L (12a)
      if (folderId && (cliente.includes(queryLower) || denom.includes(queryLower) || queryLower.includes(cliente))) {
        return { folderId, folderName: row[1] || row[0] || 'cantiere' }
      }
    }
    return null
  } catch (err) {
    console.error('findCantiereFolder error:', err)
    return null
  }
}
```

```ts
import { DRIVE_FOLDERS, findCantiereFolder, findClienteFolder, createFolder } from './drive'
import { google } from 'googleapis'

export type DocumentType = 'pos' | 'preventivo' | 'cme' | 'perizia' | 'relazione' | 'scia' | 'cila' | 'altro'

export function inferDocumentType(htmlContent: string, userPrompt: string): DocumentType {
  const text = (htmlContent + ' ' + userPrompt).toLowerCase()
  if (/\bpiano\s+operativo|\bpos\b/i.test(text)) return 'pos'
  if (/\bcomputo\s+metric|\bcme\b/i.test(text)) return 'cme'
  if (/\bperizi/i.test(text)) return 'perizia'
  if (/\brelazione\s+(?:di\s+)?calcol|\brelazione\s+tecnic/i.test(text)) return 'relazione'
  if (/\bscia\b|segnalazione\s+certif/i.test(text)) return 'scia'
  if (/\bcila\b/i.test(text)) return 'cila'
  if (/\bpreventiv/i.test(text)) return 'preventivo'
  return 'altro'
}

interface ResolvedFolder {
  folderId: string
  folderPath: string
  isFallback: boolean
}

const FALLBACK_FOLDERS: Record<DocumentType, string> = {
  pos: DRIVE_FOLDERS.POS,
  preventivo: DRIVE_FOLDERS.STUDIO_ATTIVI, // bozze sotto Studio Tecnico
  cme: DRIVE_FOLDERS.STUDIO_ATTIVI,
  perizia: DRIVE_FOLDERS.STUDIO_ATTIVI,
  relazione: DRIVE_FOLDERS.STUDIO_ATTIVI,
  scia: DRIVE_FOLDERS.STUDIO_ATTIVI,
  cila: DRIVE_FOLDERS.STUDIO_ATTIVI,
  altro: DRIVE_FOLDERS.STUDIO_ATTIVI,
}

const FALLBACK_PATHS: Record<DocumentType, string> = {
  pos: '/POS/',
  preventivo: '/Studio Tecnico ATTIVI/_Bozze/',
  cme: '/Studio Tecnico ATTIVI/_Bozze/',
  perizia: '/Studio Tecnico ATTIVI/_Bozze/',
  relazione: '/Studio Tecnico ATTIVI/_Bozze/',
  scia: '/Studio Tecnico ATTIVI/_Bozze/',
  cila: '/Studio Tecnico ATTIVI/_Bozze/',
  altro: '/Studio Tecnico ATTIVI/_Bozze/',
}

export async function resolveTargetFolder(
  docType: DocumentType,
  userPrompt: string,
  recentHistory: string,
): Promise<ResolvedFolder> {
  const context = userPrompt + ' ' + recentHistory
  // Estrai possibile nome cantiere/cliente da pattern noti
  // Pattern semplificato: "per cantiere XXX", "cliente XXX", parole capitalizzate
  const cantiereMatch = context.match(/(?:cantiere|per)\s+([A-ZÀ-Ý][\w'\s]{2,40}?)(?:\s+(?:in|a|il|la|del|via)|$|,|\.)/i)
  const clienteMatch = context.match(/(?:cliente|signor[ae]?|sig\.?)\s+([A-ZÀ-Ý][\w'\s]{2,40}?)(?:\s+|$|,|\.)/i)
  const queryName = (cantiereMatch?.[1] || clienteMatch?.[1] || '').trim()

  if (queryName.length >= 3) {
    // POS/SCIA/CILA → cerca in CANTIERI ATTIVI via REGISTRO
    if (['pos', 'scia', 'cila'].includes(docType)) {
      const found = await findCantiereFolder(queryName)
      if (found) {
        // Crea sottocartella per tipo documento (POS, Pratiche, ecc.)
        const subFolderName = docType === 'pos' ? 'POS' : 'Pratiche'
        // TODO: cerca o crea sotto-cartella; per semplicità qui salviamo direttamente in folder cantiere
        return {
          folderId: found.folderId,
          folderPath: `/CANTIERI ATTIVI/${found.folderName}/`,
          isFallback: false,
        }
      }
    }
    // Preventivi/CME/Perizie/Relazioni → cerca in Studio Tecnico ATTIVI
    if (['preventivo', 'cme', 'perizia', 'relazione'].includes(docType)) {
      const found = await findClienteFolder(queryName)
      if (found) {
        return {
          folderId: found.folderId,
          folderPath: `/Studio Tecnico ATTIVI/${found.folderName}/`,
          isFallback: false,
        }
      }
    }
  }

  // Fallback X
  return {
    folderId: FALLBACK_FOLDERS[docType],
    folderPath: FALLBACK_PATHS[docType],
    isFallback: true,
  }
}

interface SaveResult {
  driveUrl: string
  folderPath: string
  isFallback: boolean
  fileName: string
}

/**
 * Salva un HTML come Google Doc nella cartella matched.
 * Ritorna URL Drive + path + flag se è fallback.
 */
export async function saveDocumentToDrive(
  htmlContent: string,
  title: string,
  docType: DocumentType,
  userPrompt: string,
  recentHistory: string,
): Promise<SaveResult> {
  const target = await resolveTargetFolder(docType, userPrompt, recentHistory)
  const timestamp = new Date().toISOString().slice(0, 10)
  const fileName = `${title.replace(/[^\w\s-]/g, '').slice(0, 80)}_${timestamp}`

  // Crea Google Doc con contenuto HTML convertito a testo (preserve markup base)
  const auth = new google.auth.GoogleAuth({
    credentials: JSON.parse(process.env.GOOGLE_SERVICE_ACCOUNT_KEY || '{}'),
    scopes: ['https://www.googleapis.com/auth/drive'],
  })
  const drive = google.drive({ version: 'v3', auth })

  // Upload come Google Doc da HTML (Google converte automaticamente)
  const res = await drive.files.create({
    requestBody: {
      name: fileName,
      mimeType: 'application/vnd.google-apps.document',
      parents: [target.folderId],
    },
    media: {
      mimeType: 'text/html',
      body: htmlContent,
    },
    fields: 'id, name, webViewLink',
  })

  return {
    driveUrl: res.data.webViewLink || '',
    folderPath: target.folderPath,
    isFallback: target.isFallback,
    fileName: res.data.name || fileName,
  }
}
```
